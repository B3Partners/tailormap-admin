package nl.tailormap.viewer.admin.stripes;

import net.sourceforge.stripes.action.ActionBean;
import net.sourceforge.stripes.action.ActionBeanContext;
import net.sourceforge.stripes.action.ForwardResolution;
import net.sourceforge.stripes.action.Resolution;
import net.sourceforge.stripes.action.StrictBinding;
import net.sourceforge.stripes.action.UrlBinding;
import nl.tailormap.viewer.config.security.Group;

import javax.annotation.security.RolesAllowed;

/**
 *
 * @author mprins
 */
@UrlBinding("/action/about")
@StrictBinding
@RolesAllowed({Group.ADMIN})
public class AboutActionBean implements ActionBean {

    private ActionBeanContext context;

    public ActionBeanContext getContext() {
        return context;
    }

    public void setContext(ActionBeanContext context) {
        this.context = context;
    }

    public Resolution index() {
        return new ForwardResolution("/WEB-INF/jsp/about.jsp");
    }
}
